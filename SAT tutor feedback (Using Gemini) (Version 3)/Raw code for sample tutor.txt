let currentQuestionIndex = 0;

function renderQuestion() {
  const q = satData[currentQuestionIndex];
  const card = document.getElementById('question-card');
  
  // Update progress tracker
  document.getElementById('progress-tracker').innerText = `Question ${currentQuestionIndex + 1} of ${satData.length}`;

  card.innerHTML = `
    <div class="question-metadata">
      <span>${q.section}</span>
      <span>•</span>
      <span>${q.skill}</span>
    </div>
    <div class="question-text">${q.question.replace(/\n\n/g, '<br><br>')}</div>
    <div class="choices" id="choices-container">
      ${Object.entries(q.choices).map(([key, text]) => `
        <button class="choice-btn" data-key="${key}" onclick="handleAnswer('${key}')">
          <strong>${key}.</strong> ${text}
        </button>
      `).join('')}
    </div>
  `;

  document.getElementById('tutor-feedback').style.display = 'none';
  document.getElementById('next-btn').style.display = 'none';
}

async function handleAnswer(selectedKey) {
  const q = satData[currentQuestionIndex];
  const isCorrect = selectedKey === q.correctAnswer;
  
  // Lock choices and apply visual cues
  const buttons = document.querySelectorAll('.choice-btn');
  buttons.forEach(btn => {
    btn.disabled = true;
    if (btn.dataset.key === q.correctAnswer) {
      btn.classList.add('correct-reveal');
    }
    if (btn.dataset.key === selectedKey && !isCorrect) {
      btn.classList.add('wrong');
    }
  });

  // Fetch the feedback (Simulated LLM call)
  const feedbackMsg = await simulateAIFeedback(
    isCorrect, 
    selectedKey,
    q.correctAnswer,
    q.choices[selectedKey]
  );

  document.getElementById('tutor-message').innerText = feedbackMsg;
  document.getElementById('tutor-feedback').style.display = 'block';
  document.getElementById('next-btn').style.display = 'block';
  
  // Scroll to bottom so next button is visible
  setTimeout(() => {
    document.getElementById('next-btn').scrollIntoView({ behavior: 'smooth', block: 'end' });
  }, 100);
}

function nextQuestion() {
  currentQuestionIndex++;
  if (currentQuestionIndex < satData.length) {
    renderQuestion();
  } else {
    document.getElementById('progress-tracker').style.display = 'none';
    document.getElementById('question-card').innerHTML = `<div class="question-text" style="text-align:center; font-weight:bold; font-size:1.5rem;">Diagnostic Complete! <br><br>Generating your detailed gap analysis...</div>`;
    document.getElementById('tutor-feedback').style.display = 'none';
    document.getElementById('next-btn').style.display = 'none';
  }
}

// SIMULATED AI TUTOR LOGIC FOR THE DEMO

// In production, send the exact system prompt from the previous iteration to your LLM.
// For this demo, we simulate evaluating the "closeness" of the student's answer.
async function simulateAIFeedback(isCorrect, selectedKey, correctKey, selectedText) {
  if (isCorrect) {
    const praiseBank = [
      "Spot on! You nailed the core concept perfectly.",
      "Excellent work! Your reasoning is completely aligned with the text.",
      "You got it! Great job breaking down the nuances of that question.",
      "Exactly right. You didn't fall for any of the distractors!"
    ];
    return praiseBank[Math.floor(Math.random() * praiseBank.length)];
  }

  // Calculate "distance" between the choice letters to mock the AI's tiered evaluation
  // (e.g., A=65, B=66 -> distance 1)
  const distance = Math.abs(selectedKey.charCodeAt(0) - correctKey.charCodeAt(0));
  
  // Extract a small snippet of their wrong answer to prove the AI "read" it
  const words = selectedText.split(" ");
  const textSnippet = words.length > 5 ? words.slice(0, 6).join(" ") + "..." : selectedText;

  if (distance === 1) {
    // Tier 1: Very Close
    return `So close! You were on the right track focusing on "${textSnippet}", but you barely missed a tiny nuance. Look closely at the specific wording—does the text directly support that exact detail?`;
  } 
  else if (distance === 2) {
    // Tier 2: Somewhat off
    return `You're in the right ballpark, but watch your direction! A phrase like "${textSnippet}" introduces ideas or extremes not explicitly proven in the passage. Try pointing your thinking back to the main subject of the text.`;
  } 
  else {
    // Tier 3: Way off
    return `Okay, be honest... was this a random long shot? 😜 Focusing on "${textSnippet}" takes us pretty far away from the core passage! When you're stuck, go back to the most basic facts the author presents rather than taking a flyer.`;
  }
}
